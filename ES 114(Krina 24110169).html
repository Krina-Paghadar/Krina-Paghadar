    file_content = file.read().lower()  # Convert content to lowercase to handle case insensitivity

print(f"The most frequent word is '{most_frequent_word}' with {frequency} occurrences.")

# Step 5: Count the total number of words in the file
total_words = len(words)
print(f"Total number of words in the file: {total_words}")

# Step 6: Visualize the word frequencies using a bar chart
# Prepare data for plotting
words_list = list(word_counts.keys())
counts_list = list(word_counts.values())

# Create a bar chart
plt.bar(words_list, counts_list, color='skyblue', edgecolor='black')

plt.ylabel('Frequency')

def normalize(numbers):
    min_val = min(numbers)
    max_val = max(numbers)
    
   
    normalized_data = [(x - min_val) / (max_val - min_val) for x in numbers]
    
    return normalized_data
